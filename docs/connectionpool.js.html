/**
 * @class
 * @constructor
 */
function ConnectionPool() {
    var pool = {};

    /**
     * @returns {Connection[]} pool
     */
    this.getPool = function () {
        return pool;
    };

    this.size = function () {
        var c = 0;
        for (var i in pool) {
            if (pool.hasOwnProperty(i)) {
                c++;
            }
        }
        return c;
    };
}

/**
 * @param {Websocket} connection
 * @param {String} address
 * @returns {Connection}
 */
ConnectionPool.prototype.add = function (connection, address) {
    var pool = this.getPool();

    pool[address] = new Connection(connection, this);

    return pool[address];
};
/**
 * @param {String} address
 * @returns {Connection}
 */
ConnectionPool.prototype.remove = function (address) {
    var pool = this.getPool();

    if (this.has(address)) {
        var ret = pool[address];
        delete pool[address];
        return ret;
    }
};
/**
 * @param {String} address
 * @returns {bool}
 */
ConnectionPool.prototype.has = function (address) {
    var pool = this.getPool();

    return pool.hasOwnProperty(address);
};
/**
 * @param {Connection} connection
 * @returns {String}
 */
ConnectionPool.prototype.index = function (connection) {
    var pool = this.getPool();

    for (var i in pool) {
        if (!pool.hasOwnProperty(i)) {
            if (!pool.hasOwnProperty(i)) {
                continue;
            }

            if (pool[i] === connection) {
                return i;
            }
        }
    }

    return null;
};
/**
 * @param {String|Object} message
 * @param {Connection} except
 */
ConnectionPool.prototype.send = function (message, except) {
    var pool = this.getPool();

    if (typeof message !== 'string') {
        message = JSON.stringify(message);
    }

    for (var i in pool) {
        if (pool.hasOwnProperty(i)) {

            if (pool[i] === except) {
                continue;
            }

            // Connection::send(message)
            pool[i].send(message);
        }
    }
};

/**
 * @class
 * @constructor
 * @param {Websocket} socket_connection
 * @param {ConnectionPool} pool
 */
function Connection(socket_connection, pool) {
    this.conn = socket_connection;
    this.pool = pool;
}
/**
 * @returns {String}
 */
Connection.prototype.address = function () {
    return this.pool.index(this);
};
/**
 * @returns {Connection}
 */
Connection.prototype.remove = function () {
    var index = this.address();
    this.pool.remove(index);
};
/**
 * @param {String|object} message
 */
Connection.prototype.send = function (message) {
    if (typeof message !== 'string') {
        message = JSON.stringify(message);
    }
    this.conn.send(message);
};
module.exports = ConnectionPool;
