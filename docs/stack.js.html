/**
 * @class
 * @constructor
 * @param {Number} size
 * @param {bool} useHash Use hashCode to calculate differences, instead of entire object
 */
function Stack(size, useHash) {
    var maxLength = size || 10;
    var elements = [];
    var hash = useHash || false;

    /**
     * @param {Object} elem1
     * @param {Object} elem2
     * @returns {bool}
     */
    this.compare = function (elem1, elem2) {
        if (hash) {
            return elem1.__hashCode === elem2.__hashCode;
        }
        return elem1 === elem2;
    };
    /**
     * @param {Object} elem
     * @returns {Object|null}
     */
    this.add = function (elem) {
        elements.push(elem);

        if (elements.length >= maxLength) {
            return elements.shift();
        }
    };
    /**
     * @param {Object|Number} elem
     * @returns {Object}
     */
    this.remove = function (elem) {
        if (typeof elem === 'number') {
            return elements.splice(elem, 1)[0];
        } else {
            var index = this.index(elem);
            if (index >= 0) {
                return elements.splice(index, 1)[0];
            }
        }
        return null;
    };
    /**
     * @param {Object} elem
     * @returns {Number}
     */
    this.index = function (elem) {
        for (var i = 0; i < elements.length; i++) {
            if (this.compare(elements[i], elem)) {
                return i;
            }
        }
        return -1;
    };
    /**
     * @param {Object} elem
     * @returns {bool}
     */
    this.contains = function (elem) {
        return this.index(elem) > -1;
    };
    /**
     * @param {Number} index
     * @returns {Object}
     */
    this.get = function (index) {
        if (index >= 0 && index < elements.length) {
            return elements[index];
        }
        return null;
    };
    /**
     * @returns {Number}
     */
    this.size = function () {
        return elements.length;
    };
    /**
     * @returns {Number}
     */
    this.capacity = function () {
        return maxLength;
    };
}

module.exports = Stack;
