var restify = require('restify');

/*******************************************
 * Basic API setup
 *
 * GET    file/         - listing of files (maybe not?)
 * GET    file/<name>   - get file <name>
 * PUT    file/<name>   - upload file to <name>
 * DELETE file/<name>   - delete file <name>
 *
 * GET    statistics    - get server stats
 */

/**
 * For more documentation of what it does {@link http://mcavage.me/node-restify/}
 *
 * @class
 * @constructor
 * @param {Server} main
 * @param {Object} config
 */
function APIServer(main, config) {
    var self = this;
    this.config = config;
    this.main = main;

    // @todo continue

    /**
     * @function
     */
    this.init = function () {

        var server = restify.createServer({
            name: self.config.address,
            version: '0.0.1'
        });
        self.server = server;

        server.use(restify.plugins.acceptParser(server.acceptable));
        server.use(restify.plugins.queryParser());
        server.use(restify.plugins.bodyParser({
            mapParams: true
        }));
        restify.defaultResponseHeaders = function () {
            this.header('X-Server', 'scalable-server v3.1.0');
            this.header('Accepts', 'application/json');
            this.header('Content-Type', 'application/json');
        };

        server.pre(restify.pre.userAgentConnection());

        // setup api calls
        server.get('/statistics', function (req, res, next) {
            //
            var data = self.main.statistics();

            res.json(200, data);

            return next();
        });

        server.listen(self.config.port, function () {
            self.main.log('APIServer', '%s listening at %s', server.name, server.url);
        });
    };
}

module.exports = APIServer;
