var restifyClients = require('restify-clients');

/**
 * @class
 * @constructor
 * @param {Client} main
 * @param {String} address
 */
function APIClient(main, address) {
    this.api_client = restifyClients.createJsonClient({
        url: address,
        version: '*'
    });

    this.main = main;
}

/**
 * @function
 * @param {string} path
 * @param {Function} cb
 */
APIClient.prototype.get = function (path, cb) {
    var self = this;

    this.api_client.get(path, function (err, req, res, obj) {
        if (err) {
            self.main.emit('error', err);
        } else {
            cb.apply(self, [obj]);
        }
    });
};
